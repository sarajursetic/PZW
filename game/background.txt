const drawMap = () => {
  for (let eachRow = 0; eachRow < gridRows; eachRow++) {
    for (let eachCol = 0; eachCol < gridCols; eachCol++) {
     let arrayIndex = eachRow * gridCols + eachCol;

      if (map[arrayIndex] === 1) {
        ctx.fillStyle = "lightgray";
      } else {
        ctx.fillStyle = "black";
      }

      ctx.fillRect(tileW * eachCol, tileH * eachRow, tileW, tileH);
    }
  }
}
